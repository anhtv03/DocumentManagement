    /**
     * Handles the HTTP <code>POST</code> method.

     * @param request servlet request
     * @param response servlet response

    @Override
    protected void doPost(HttpServletRequest request, HttpServletResponse response)
            throws ServletException, IOException {
         DAO dao = new DAO();
        ArrayList<User> al = dao.getalluser();
        String name = request.getParameter("name");
        String pass = request.getParameter("pass");
        String mess = "Login Failed";
        for (User u : al) {
            if (u.getName().equals(name) && u.getPass().equals(pass)) {
                mess = "Login successfull!";
                HttpSession session = request.getSession();
                session.setAttribute("u", u);
            }
        }
        request.setAttribute("mess", mess);
        request.getRequestDispatcher("login.jsp").forward(request, response);
    }

package dal;

import java.sql.PreparedStatement;
import java.sql.ResultSet;
import java.util.ArrayList;
import model.Article;
import model.User;

/**
 *
 * @author Admin
 */
public class DAO extends DBContext {

    public ArrayList<User> getalluser() {
        ArrayList<User> list = new ArrayList<>();
        try {
            String sql = "  select * from [User]";
            PreparedStatement stm = connection.prepareStatement(sql);
            ResultSet rs = stm.executeQuery();
            while (rs.next()) {
                User s = new User(rs.getString(1), rs.getString(2));
                list.add(s);
            }
        } catch (Exception e) {
        }
        return list;
    }

    public Article getallarticle(String aid) {
        try {
            String sql = "   select * from Article where aid = " + aid;
            PreparedStatement stm = connection.prepareStatement(sql);
            ResultSet rs = stm.executeQuery();
            while (rs.next()) {
                Article s = new Article(rs.getInt(1), rs.getString(2), rs.getDate(3), rs.getBoolean(4));
                return s;
            }
        } catch (Exception e) {
        }
        return null;
    }

}

package model;

import java.sql.Date;

/**
 *
 * @author Admin
 */
public class Article {

    int aid;
    String title;
    Date published_date;
    boolean open_access;

    public Article(int aid, String title, Date published_date, boolean open_access) {
        this.aid = aid;
        this.title = title;
        this.published_date = published_date;
        this.open_access = open_access;
    }

    public int getAid() {
        return aid;
    }

    public void setAid(int aid) {
        this.aid = aid;
    }

    public String getTitle() {
        return title;
    }

    public void setTitle(String title) {
        this.title = title;
    }

    public Date getPublished_date() {
        return published_date;
    }

    public void setPublished_date(Date published_date) {
        this.published_date = published_date;
    }

    public boolean isOpen_access() {
        return open_access;
    }

    public void setOpen_access(boolean open_access) {
        this.open_access = open_access;
    }

    public Article() {
    }

}
